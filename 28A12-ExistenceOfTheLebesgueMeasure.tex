\begin{theorem}[Lebesgue]
Let $\mathcal{B}$ be the \PMlinkname{Borel $\sigma$-algebra}{BorelSigmaAlgebra} on the real number line. Then, there is a unique measure $\mu$ on the measurable space $(\mathbb{R},\mathcal{B})$ satisfying
\begin{equation*}
\mu\left( (a,b) \right) = b-a
\end{equation*}
for all real numbers $a<b$.
\end{theorem}
